1Modèle de données & périmètre

Le modèle relationnel retenu (validé tech lead / PO)

schéma corpo (PostgreSQL) ├── corpo_denomination ← RÉFÉRENTIEL : les 11 coupures de la Jira (seedées) │ code unique (BILLET_200…), item_type, face_value, label ├── corpo_cash_request ← ENTÊTE : réf, type, STATUT, client (zones libres), │ montant, urgent, dates INITIATED / VALIDATED / CANCELLED └── corpo_request_denomination ← VENTILATION : (demande, coupure) → quantité FK vers l'entête (CASCADE) et vers le référentiel (RESTRICT), unicité (demande, coupure)

La ventilation n'est plus 11 colonnes plates mais des lignes normalisées pointant vers un référentiel : ajouter demain une coupure = une ligne de seed, zéro migration de structure. Les 11 codes du référentiel correspondent exactement aux champs de la Jira : BILLET_200/100/50/20, PIECE_10/5/1, CENTIME_50/20/10/5. Le référentiel est propre au schéma corpo (pas de FK cross-schema vers public.denomination, dont le contenu du dump ne couvre pas les centimes) — question n°3 si le tech lead préfère mutualiser.

-- Jira AER_LM-257 — Modele de donnees "Demandes Corpo" (RELATIONNEL + STATUT)
-- Schema PostgreSQL dedie "corpo" :
--   corpo.corpo_denomination         referentiel des 11 coupures (seed inclus)
--   corpo.corpo_cash_request         entete de la demande (+ statut)
--   corpo.corpo_request_denomination ventilation (N lignes par demande)
--   corpo.corpo_request_ref_seq      sequence de la "Ref demande"
-- A executer dans pgAdmin (Query Tool sur qlick_flow) AVANT le premier
-- demarrage : le schema doit exister, la sequence et le SEED du referentiel
-- ne seront jamais crees par ddl-auto.
-- ============================================================================

CREATE SCHEMA IF NOT EXISTS corpo;

-- 1. Referentiel des coupures (les 11 de la Jira, en MAD)
CREATE TABLE corpo.corpo_denomination (
    id                 uuid PRIMARY KEY,
    creation_date      timestamp,
    modification_date  timestamp,
    denomination_code  varchar(20) NOT NULL UNIQUE,   -- ex: BILLET_200
    item_type          varchar(10) NOT NULL,          -- BANKNOTE / COIN
    face_value         numeric(10,2) NOT NULL,        -- 200.00 ... 0.05
    denomination_label varchar(100) NOT NULL,
    CONSTRAINT chk_corpo_item_type CHECK (item_type IN ('BANKNOTE', 'COIN'))
);

-- 2. Entete de la demande Corporate
CREATE TABLE corpo.corpo_cash_request (
    id                 uuid PRIMARY KEY,
    creation_date      timestamp,                     -- "Date demande"
    modification_date  timestamp,                     -- "Date mise a jour demande"
    unique_request_ref varchar(30) UNIQUE,            -- "Ref demande" (CO-AAAA-NNNNNN)
    request_type       varchar(20) NOT NULL,          -- ALIMENTATION / RAMASSAGE
    request_status     varchar(30) NOT NULL DEFAULT 'INITIATED',
    client_ref         varchar(50),                   -- zones libres (Jira)
    city               varchar(100),
    address_complement varchar(255),
    account_number     varchar(34),
    amount             numeric(15,2) NOT NULL DEFAULT 0,
    is_urgent          varchar(3) NOT NULL DEFAULT 'NO',
    CONSTRAINT chk_corpo_request_type   CHECK (request_type IN ('ALIMENTATION', 'RAMASSAGE')),
    CONSTRAINT chk_corpo_request_status CHECK (request_status IN ('INITIATED', 'VALIDATED', 'CANCELLED')),
    CONSTRAINT chk_corpo_is_urgent      CHECK (is_urgent IN ('YES', 'NO'))
);

-- 3. Ventilation : quantite par coupure pour une demande
CREATE TABLE corpo.corpo_request_denomination (
    id                 uuid PRIMARY KEY,
    creation_date      timestamp,
    modification_date  timestamp,
    corpo_request_id   uuid NOT NULL,
    denomination_id    uuid NOT NULL,
    quantity           integer NOT NULL DEFAULT 0,    -- "Valeur par defaut 0" (Jira)
    CONSTRAINT fk_corpo_rd_request      FOREIGN KEY (corpo_request_id) REFERENCES corpo.corpo_cash_request(id) ON DELETE CASCADE,
    CONSTRAINT fk_corpo_rd_denomination FOREIGN KEY (denomination_id)  REFERENCES corpo.corpo_denomination(id) ON DELETE RESTRICT,
    CONSTRAINT uq_corpo_rd UNIQUE (corpo_request_id, denomination_id),
    CONSTRAINT chk_corpo_rd_quantity CHECK (quantity >= 0)
);

-- 4. Sequence de la reference lisible
CREATE SEQUENCE corpo.corpo_request_ref_seq START WITH 1 INCREMENT BY 1;

/**
 * Agregat "demande Corporate" — Jira AER_LM-257, modele RELATIONNEL (valide
 * tech lead / PO) : entete + ventilation par coupures du referentiel corpo,
 * et cycle de vie INITIATED / VALIDATED / CANCELLED comme le nominal.
 *
 * Montant : "Valeur par defaut 0 - en cas de ramassage, il sera alimente en
 * direct" (Jira). ALIMENTATION : montant deduit de la ventilation
 * (somme valeur faciale x quantite, faciales issues du referentiel) ;
 * RAMASSAGE : montant pris tel que fourni (defaut 0).
 */
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class CorpoCashRequest extends AggregateRoot<CorpoCashRequestId> {

    public static final String STATUS_INITIATED = "INITIATED";
    public static final String STATUS_VALIDATED = "VALIDATED";
    public static final String STATUS_CANCELLED = "CANCELLED";

    private String uniqueRequestRef;
    private CorpoRequestType requestType;
    private String requestStatus = STATUS_INITIATED;
    private String clientRef;
    private String city;
    private String addressComplement;
    private String accountNumber;
    private List<CorpoRequestDenomination> denominations = new ArrayList<>();
    private BigDecimal amount = BigDecimal.ZERO;
    private String isUrgent = "NO";
    private LocalDateTime createdAt;
    private LocalDateTime updatedAt;

    /**
     * Applique les defauts de la Jira (quantites 0, montant 0, urgent NO,
     * statut INITIATED) et calcule le montant d'une ALIMENTATION depuis la
     * ventilation. faceValuesByCode = referentiel corpo (code -> valeur
     * faciale), fourni par le cas d'usage — meme pattern que le nominal.
     */
    public void validateAndComputeAmount(Map<String, BigDecimal> faceValuesByCode) {
        if (requestType == null) {
            throw new IllegalStateException(
                    "Le type de demande (ALIMENTATION/RAMASSAGE) est obligatoire");
        }
        if (requestStatus == null) {
            requestStatus = STATUS_INITIATED;
        }
        if (isUrgent == null) {
            isUrgent = "NO";
        }
        if (denominations == null) {
            denominations = new ArrayList<>();
        }
        denominations.forEach(line -> {
            if (line.getQuantity() == null) {
                line.setQuantity(0); // defaut Jira
            }
        });

        if (requestType == CorpoRequestType.ALIMENTATION) {
            BigDecimal total = BigDecimal.ZERO;
            for (CorpoRequestDenomination line : denominations) {
                BigDecimal faceValue = faceValuesByCode.get(line.getDenominationCode());
                if (faceValue == null) {
                    throw new IllegalArgumentException(
                            "Coupure inconnue : " + line.getDenominationCode());
                }
                total = total.add(faceValue.multiply(BigDecimal.valueOf(line.getQuantity())));
            }
            this.amount = total;
        } else if (this.amount == null) {
            this.amount = BigDecimal.ZERO; // defaut Jira, "alimente en direct" sinon
        }
    }
}

            "BILLET_200", new BigDecimal("200.00"),
            "BILLET_100", new BigDecimal("100.00"),
            "PIECE_5", new BigDecimal("5.00"),

    private CorpoRequestDenomination line(String code, Integer quantity) {

    @Test

        request.setRequestType(CorpoRequestType.ALIMENTATION);

 * memes gardes que le nominal. Not-found en RuntimeException (convention
 * actuelle du service).

@Component
@Slf4j
@RequiredArgsConstructor
public class CorpoDemandRepositoryImpl implements CorpoDemandRepository {

    private final CorpoCashRequestJpaRepository corpoJpaRepository;
    private final CorpoDenominationJpaRepository corpoDenominationJpaRepository;
    private final CorpoCashRequestPersistenceMapper mapper;

    @Override
    @Transactional
    public CorpoCashRequest save(CorpoCashRequest request) {
        if (request.getUniqueRequestRef() == null) {
            Long next = corpoJpaRepository.nextRequestRef();
            request.setUniqueRequestRef(
                    "CO-" + Year.now().getValue() + "-" + String.format("%06d", next));
        }
        CorpoCashRequestEntity entity = mapper.toPersistence(request, denominationsByCode());
        return mapper.toDomain(corpoJpaRepository.save(entity));
    }

    @Override
    @Transactional(readOnly = true)
    public CorpoCashRequest getById(UUID id) {
        CorpoCashRequestEntity entity = corpoJpaRepository.findByIdWithDenominations(id)
                .orElseThrow(() -> new RuntimeException("Corpo cash demand not found"));
        return mapper.toDomain(entity);
    }

    @Override
    @Transactional(readOnly = true)
    public List<CorpoCashRequest> findAll() {
        return corpoJpaRepository.findAllWithDenominations().stream()
                .map(mapper::toDomain)
                .toList();
    }

    @Override
    @Transactional
    public CorpoCashRequest update(UUID id, CorpoCashRequest request) {
        CorpoCashRequestEntity existing = corpoJpaRepository.findByIdWithDenominations(id)
                .orElseThrow(() -> new RuntimeException("Corpo cash demand not found"));
        if ("CANCELLED".equals(existing.getRequestStatus())) {
            throw new RuntimeException("Cannot update cancelled corpo cash demand");
        }
        mapper.applyHeaderFields(existing, request);
        existing.getDenominations().clear();
        existing.getDenominations().addAll(
                mapper.toDenominationEntities(request, existing, denominationsByCode()));
        return mapper.toDomain(corpoJpaRepository.save(existing));
    }

    @Override
    @Transactional
    public CorpoCashRequest cancelById(UUID id) {
        CorpoCashRequestEntity existing = corpoJpaRepository.findByIdWithDenominations(id)
                .orElseThrow(() -> new RuntimeException("Corpo cash demand not found"));
        if ("CANCELLED".equals(existing.getRequestStatus())) {
            throw new RuntimeException("Corpo cash demand already cancelled");
        }
        existing.setRequestStatus("CANCELLED");
        return mapper.toDomain(corpoJpaRepository.save(existing));
    }

    private Map<String, CorpoDenominationEntity> denominationsByCode() {
        return corpoDenominationJpaRepository.findAll().stream()
                .collect(Collectors.toMap(
                        CorpoDenominationEntity::getDenominationCode, d -> d));
    }
}

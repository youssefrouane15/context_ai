        if (currencyDetails == null) currencyDetails = new ArrayList<>();

                        faceValue.multiply(BigDecimal.valueOf(line.getQuantity())));
            }
            detail.setTotalAmountByCurrency(detailTotal);
            computed = computed.add(detailTotal);

1Vue d'ensemble : ce qui change et pourquoi

Le nouveau contrat d'erreur

CasHTTPCode APIExemple de message
Corps invalide (@Valid) ou JSON/enum illisible400CORPO_BAD_REQUEST« Un ou plusieurs champs sont invalides » + détail par champ
Demande introuvable404CORPO_NOT_FOUND« La demande corpo <id> est introuvable »
État incompatible409CORPO_CONFLICT« La demande CO-2026-000001 est déjà annulée »
Règle métier violée422CORPO_VALIDATION_ERROR« Coupure inconnue : BILLET_500 »

Format : ProblemDetail (RFC 7807), avec propriétés code (lisible machine, pour le front) et timestamp. Trois mécanismes s'enchaînent : les exceptions typées naissent dans le domaine/l'adaptateur avec un message métier ; le use case ne les attrape plus (les try/catch + SessionServiceException écrasaient tout en 500 — c'est le prérequis de tout le lot) ; le CorpoDemandExceptionHandler les traduit en HTTP.

Point de design important : le handler est scopé @RestControllerAdvice(assignableTypes = CorpoDemandController.class) — il n'intercepte QUE les endpoints corpo. Le nominal (/cash) garde exactement son comportement actuel (SessionServiceException → 500). La généralisation au nominal viendra avec l'annexe correctifs, pas en douce ici.

ChantierDétail

package com.bmci.q360.domain.corpo;

import com.bmci.q360.domain.corpo.enums.CorpoRequestStatus;
import com.bmci.q360.domain.corpo.enums.CorpoRequestType;
import com.bmci.q360.domain.corpo.enums.UrgencyFlag;
import com.bmci.q360.domain.corpo.exceptions.CorpoValidationException;
import com.bmci.q360.domain.corpo.valueobjects.CorpoCashRequestId;
import com.bmci.q360.domain.entity.common.AggregateRoot;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;

/**
 * Agregat "demande Corporate" — Jira AER_LM-257.
 * Statut et flag urgent portes par des enums (plus de magic strings), regles
 * metier violees signalees par CorpoValidationException (-> 422 cote API).
 */
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class CorpoCashRequest extends AggregateRoot<CorpoCashRequestId> {

    private String uniqueRequestRef;
    private CorpoRequestType requestType;
    private CorpoRequestStatus requestStatus = CorpoRequestStatus.INITIATED;
    private String clientRef;
    private String city;
    private String addressComplement;
    private String accountNumber;
    private List<CorpoRequestDenomination> denominations = new ArrayList<>();
    private BigDecimal amount = BigDecimal.ZERO;
    private UrgencyFlag isUrgent = UrgencyFlag.NO;
    private LocalDateTime createdAt;
    private LocalDateTime updatedAt;

    /**
     * Applique les defauts de la Jira et calcule le montant d'une
     * ALIMENTATION depuis la ventilation (faciales fournies par le
     * referentiel corpo, cle = code coupure).
     */
    public void validateAndComputeAmount(Map<String, BigDecimal> faceValuesByCode) {
        if (requestType == null) {
            throw new CorpoValidationException(
                    "Le type de demande (ALIMENTATION/RAMASSAGE) est obligatoire");
        }
        if (requestStatus == null) {
            requestStatus = CorpoRequestStatus.INITIATED;
        }
        if (isUrgent == null) {
            isUrgent = UrgencyFlag.NO;
        }
        if (denominations == null) {
            denominations = new ArrayList<>();
        }
        denominations.forEach(line -> {
            if (line.getQuantity() == null) {
                line.setQuantity(0); // defaut Jira
            }
        });

        if (requestType == CorpoRequestType.ALIMENTATION) {
            BigDecimal total = BigDecimal.ZERO;
            for (CorpoRequestDenomination line : denominations) {
                BigDecimal faceValue = faceValuesByCode.get(line.getDenominationCode());
                if (faceValue == null) {
                    throw new CorpoValidationException(
                            "Coupure inconnue : " + line.getDenominationCode());
                }
                total = total.add(faceValue.multiply(BigDecimal.valueOf(line.getQuantity())));
            }
            this.amount = total;
        } else if (this.amount == null) {
            this.amount = BigDecimal.ZERO; // defaut Jira, "alimente en direct" sinon
        }
    }
}

package com.bmci.workflow.fund.management.service.domain.corpo.ports;

import com.bmci.q360.domain.corpo.CorpoCashRequest;
import com.bmci.workflow.fund.management.service.domain.corpo.dto.CorpoCashRequestDTO;
import com.bmci.workflow.fund.management.service.domain.corpo.mapper.CorpoCashRequestWebMapper;
import com.bmci.workflow.fund.management.service.domain.corpo.ports.input.CorpoDemandUseCase;
import com.bmci.workflow.fund.management.service.domain.corpo.ports.output.CorpoDemandRepository;
import com.bmci.workflow.fund.management.service.domain.corpo.ports.output.CorpoDenominationRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.math.BigDecimal;
import java.util.List;
import java.util.Map;
import java.util.UUID;

/**
 * Cas d'usage des demandes Corporate — Jira AER_LM-257.
 * Plus de try/catch generique ni de SessionServiceException : les exceptions
 * metier typees (CorpoValidationException, CorpoNotFoundException,
 * CorpoConflictException) remontent librement jusqu'au
 * CorpoDemandExceptionHandler qui les traduit en 422 / 404 / 409 avec un
 * message metier explicite. Attraper ici les aurait toutes ecrasees en 500.
 */
@RequiredArgsConstructor
@Service
public class CorpoDemandUseCaseImpl implements CorpoDemandUseCase {

    private final CorpoDemandRepository corpoDemandRepository;
    private final CorpoDenominationRepository corpoDenominationRepository;
    private final CorpoCashRequestWebMapper webMapper;

    @Override
    @Transactional
    public CorpoCashRequestDTO create(CorpoCashRequestDTO dto) {
        CorpoCashRequest request = webMapper.toDomain(dto);
        Map<String, BigDecimal> faceValues = corpoDenominationRepository.findFaceValuesByCode();
        request.validateAndComputeAmount(faceValues);
        return webMapper.toResponseDTO(corpoDemandRepository.save(request));
    }

    @Override
    public CorpoCashRequestDTO getById(UUID id) {
        return webMapper.toResponseDTO(corpoDemandRepository.getById(id));
    }

    @Override
    public List<CorpoCashRequestDTO> getAll() {
        return corpoDemandRepository.findAll().stream()
                .map(webMapper::toResponseDTO)
                .toList();
    }

    @Override
    @Transactional
    public CorpoCashRequestDTO update(UUID id, CorpoCashRequestDTO dto) {
        CorpoCashRequest request = webMapper.toDomain(dto);
        Map<String, BigDecimal> faceValues = corpoDenominationRepository.findFaceValuesByCode();
        request.validateAndComputeAmount(faceValues);
        return webMapper.toResponseDTO(corpoDemandRepository.update(id, request));
    }

    @Override
    @Transactional
    public CorpoCashRequestDTO cancel(UUID id) {
        return webMapper.toResponseDTO(corpoDemandRepository.cancelById(id));
    }
}

import com.bmci.q360.domain.corpo.CorpoCashRequest;

import org.mapstruct.Mapper;

import org.mapstruct.MappingTarget;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.UUID;
import java.util.stream.Collectors;

/**
 * Mapper de persistance corpo — HYBRIDE MapStruct : les mappings mecaniques
 * (entite -> domaine, champs d'entete) sont generes ; la logique qui ne se
 * genere pas (resolution du referentiel par code, cablage du parent,
 * generation de l'arbre a la creation) reste en methodes manuelles.
 * Enums <-> varchar geres automatiquement par MapStruct (name()/valueOf).
 */
@Mapper(componentModel = "spring")
public abstract class CorpoCashRequestPersistenceMapper {

    // ================= GENERE PAR MAPSTRUCT =================

    /** ENTITE -> DOMAINE (lecture) : entierement genere. */
    @Mapping(target = "createdAt", source = "creationDate")
    @Mapping(target = "updatedAt", source = "modificationDate")
    public abstract CorpoCashRequest toDomain(CorpoCashRequestEntity entity);

    /** Ligne de ventilation : aplatit le referentiel dans le domaine. */
    @Mapping(target = "denominationCode", source = "denomination.denominationCode")
    @Mapping(target = "denominationLabel", source = "denomination.denominationLabel")
    @Mapping(target = "itemType", source = "denomination.itemType")
    @Mapping(target = "faceValue", source = "denomination.faceValue")
    protected abstract CorpoRequestDenomination toDenominationDomain(
            CorpoRequestDenominationEntity entity);

    /**
     * Champs d'entete modifiables (creation ET update) : genere, en ignorant
     * ce que le client ne pilote pas (id, ref, statut, dates, ventilation).
     */
    @Mapping(target = "id", ignore = true)
    @Mapping(target = "creationDate", ignore = true)
    @Mapping(target = "modificationDate", ignore = true)
    @Mapping(target = "uniqueRequestRef", ignore = true)
    @Mapping(target = "requestStatus", ignore = true)
    @Mapping(target = "denominations", ignore = true)
    public abstract void applyHeaderFields(
            @MappingTarget CorpoCashRequestEntity entity, CorpoCashRequest domain);

    // ================= LOGIQUE MANUELLE =================

    /** DOMAINE -> ENTITE (creation) : orchestration manuelle. */
    public CorpoCashRequestEntity toPersistence(
            CorpoCashRequest domain,
            Map<String, CorpoDenominationEntity> denominationsByCode) {
        if (domain == null) return null;

        CorpoCashRequestEntity entity = new CorpoCashRequestEntity();
        if (domain.getId() != null && domain.getId().getValue() != null) {
            entity.setId(domain.getId().getValue());
        }
        entity.setUniqueRequestRef(domain.getUniqueRequestRef());
        applyHeaderFields(entity, domain);
        entity.setRequestStatus(domain.getRequestStatus() != null
                ? domain.getRequestStatus().name()
                : CorpoRequestStatus.INITIATED.name());
        entity.setDenominations(toDenominationEntities(domain, entity, denominationsByCode));
        return entity;
    }

    /** Ventilation : resolution referentiel par code + cablage du parent. */
    public List<CorpoRequestDenominationEntity> toDenominationEntities(
            CorpoCashRequest domain,
            CorpoCashRequestEntity parent,
            Map<String, CorpoDenominationEntity> denominationsByCode) {

        if (domain.getDenominations() == null) {
            return new ArrayList<>();
        }
        return domain.getDenominations().stream()
                .map(line -> {
                    CorpoRequestDenominationEntity lineEntity = new CorpoRequestDenominationEntity();
                    if (line.getId() != null && line.getId().getValue() != null) {
                        lineEntity.setId(line.getId().getValue());
                    }
                    lineEntity.setCorpoRequest(parent);
                    lineEntity.setQuantity(line.getQuantity());

                    CorpoDenominationEntity matched =
                            denominationsByCode.get(line.getDenominationCode());
                    if (matched == null) {
                        throw new CorpoValidationException(
                                "La coupure '" + line.getDenominationCode()
                                        + "' n'existe pas dans le référentiel corpo");
                    }
                    lineEntity.setDenomination(matched);
                    return lineEntity;
                })
                .collect(Collectors.toCollection(ArrayList::new));
    }

    // ----- Conversions UUID <-> value objects pour le mapping genere -----

    protected CorpoCashRequestId toCashRequestId(UUID value) {
        return value == null ? null : new CorpoCashRequestId(value);
    }

    protected CorpoRequestDenominationId toDenominationId(UUID value) {
        return value == null ? null : new CorpoRequestDenominationId(value);
    }
}

import java.time.Year;

/**
 * Adaptateur de persistance des demandes Corporate — Jira AER_LM-257.
 * Erreurs d'etat en exceptions typees a message metier explicite
 * (introuvable -> 404, deja annulee / modification d'une annulee -> 409).
 * Le flush() de update force les DELETE de l'ancienne ventilation avant les
 * INSERT de la nouvelle (Hibernate insere avant de supprimer, ce qui violait
 * la contrainte uq_corpo_rd).
 */
@Component
@RequiredArgsConstructor
public class CorpoDemandRepositoryImpl implements CorpoDemandRepository {

    private final CorpoCashRequestJpaRepository corpoJpaRepository;
    private final CorpoDenominationJpaRepository corpoDenominationJpaRepository;
    private final CorpoCashRequestPersistenceMapper mapper;

    @Override
    @Transactional
    public CorpoCashRequest save(CorpoCashRequest request) {
        if (request.getUniqueRequestRef() == null) {
            Long next = corpoJpaRepository.nextRequestRef();
            request.setUniqueRequestRef(
                    "CO-" + Year.now().getValue() + "-" + String.format("%06d", next));
        }
        CorpoCashRequestEntity entity = mapper.toPersistence(request, denominationsByCode());
        return mapper.toDomain(corpoJpaRepository.save(entity));
    }

    @Override
    @Transactional(readOnly = true)
    public CorpoCashRequest getById(UUID id) {
        return mapper.toDomain(findEntityOrThrow(id));
    }

    @Override
    @Transactional(readOnly = true)
    public List<CorpoCashRequest> findAll() {
        return corpoJpaRepository.findAllWithDenominations().stream()
                .map(mapper::toDomain)
                .toList();
    }

    @Override
    @Transactional
    public CorpoCashRequest update(UUID id, CorpoCashRequest request) {
        CorpoCashRequestEntity existing = findEntityOrThrow(id);
        if (isCancelled(existing)) {
            throw new CorpoConflictException(
                    "Impossible de modifier la demande " + existing.getUniqueRequestRef()
                            + " : elle est annulée");
        }
        mapper.applyHeaderFields(existing, request);
        existing.getDenominations().clear();
        corpoJpaRepository.flush(); // DELETE avant INSERT (contrainte uq_corpo_rd)
        existing.getDenominations().addAll(
                mapper.toDenominationEntities(request, existing, denominationsByCode()));
        return mapper.toDomain(corpoJpaRepository.save(existing));
    }

    @Override
    @Transactional
    public CorpoCashRequest cancelById(UUID id) {
        CorpoCashRequestEntity existing = findEntityOrThrow(id);
        if (isCancelled(existing)) {
            throw new CorpoConflictException(
                    "La demande " + existing.getUniqueRequestRef() + " est déjà annulée");
        }
        existing.setRequestStatus(CorpoRequestStatus.CANCELLED.name());
        return mapper.toDomain(corpoJpaRepository.save(existing));
    }

    private CorpoCashRequestEntity findEntityOrThrow(UUID id) {
        return corpoJpaRepository.findByIdWithDenominations(id)
                .orElseThrow(() -> new CorpoNotFoundException(
                        "La demande corpo " + id + " est introuvable"));
    }

    private boolean isCancelled(CorpoCashRequestEntity entity) {
        return CorpoRequestStatus.CANCELLED.name().equals(entity.getRequestStatus());
    }

    private Map<String, CorpoDenominationEntity> denominationsByCode() {
        return corpoDenominationJpaRepository.findAll().stream()
                .collect(Collectors.toMap(
                        CorpoDenominationEntity::getDenominationCode, d -> d));
    }
}

/**
 * Contrat d'erreur des endpoints corpo (RFC 7807 / ProblemDetail).
 * SCOPE VOLONTAIREMENT LIMITE au CorpoDemandController (assignableTypes) :
 * le comportement du nominal reste strictement inchange en attendant la
 * generalisation (annexe correctifs).
 *
 * Codes API :
 *   CORPO_BAD_REQUEST       400  corps invalide / champs invalides
 *   CORPO_NOT_FOUND         404  demande introuvable
 *   CORPO_CONFLICT          409  etat incompatible (deja annulee...)
 *   CORPO_VALIDATION_ERROR  422  regle metier violee (type manquant, coupure inconnue)
 */
@RestControllerAdvice(assignableTypes = CorpoDemandController.class)
public class CorpoDemandExceptionHandler {

    @ExceptionHandler(CorpoNotFoundException.class)
    public ProblemDetail handleNotFound(CorpoNotFoundException e) {
        return problem(HttpStatus.NOT_FOUND, "Demande introuvable", e.getMessage(), "CORPO_NOT_FOUND");
    }

    @ExceptionHandler(CorpoConflictException.class)
    public ProblemDetail handleConflict(CorpoConflictException e) {
        return problem(HttpStatus.CONFLICT, "Opération non permise", e.getMessage(), "CORPO_CONFLICT");
    }

    @ExceptionHandler(CorpoValidationException.class)
    public ProblemDetail handleValidation(CorpoValidationException e) {
        return problem(HttpStatus.UNPROCESSABLE_ENTITY, "Règle métier violée",
                e.getMessage(), "CORPO_VALIDATION_ERROR");
    }

    /** Violations Bean Validation (@Valid) : détail champ par champ. */
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ProblemDetail handleInvalidBody(MethodArgumentNotValidException e) {
        ProblemDetail problem = problem(HttpStatus.BAD_REQUEST, "Requête invalide",
                "Un ou plusieurs champs sont invalides", "CORPO_BAD_REQUEST");
        Map<String, String> fieldErrors = new LinkedHashMap<>();
        e.getBindingResult().getFieldErrors().forEach(fieldError ->
                fieldErrors.put(fieldError.getField(), fieldError.getDefaultMessage()));
        problem.setProperty("fieldErrors", fieldErrors);
        return problem;
    }

    /** JSON illisible ou valeur d'enum inconnue (ex. isUrgent: "PEUT-ETRE"). */
    @ExceptionHandler(HttpMessageNotReadableException.class)
    public ProblemDetail handleUnreadable(HttpMessageNotReadableException e) {
        return problem(HttpStatus.BAD_REQUEST, "Requête invalide",
                "Corps de requête illisible ou valeur non autorisée "
                        + "(vérifier requestType, isUrgent et le format JSON)",
                "CORPO_BAD_REQUEST");
    }

    private ProblemDetail problem(HttpStatus status, String title, String detail, String code) {
        ProblemDetail problem = ProblemDetail.forStatusAndDetail(status, detail);
        problem.setTitle(title);
        problem.setProperty("code", code);
        problem.setProperty("timestamp", Instant.now());
        return problem;
    }

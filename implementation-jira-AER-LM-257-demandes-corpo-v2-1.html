3Étape A — Base de données (pgAdmin)

Architecture à deux schémas — comment ça marche

Une base PostgreSQL peut contenir plusieurs schémas (des « dossiers » de tables). Ici : public garde tout le nominal, et un schéma corpo héberge la table et la séquence Corporate. Une seule base, une seule datasource, aucune configuration supplémentaire — voici pourquoi, point par point :

AspectDétail
application.yamlAucun changement. L'URL JDBC pointe sur la base, pas sur un schéma ; le ciblage se fait au niveau des entités
Entité JPA@Table(name = "corpo_cash_request", schema = "corpo") → Hibernate qualifie toutes ses requêtes en corpo.corpo_cash_request (INSERT, SELECT, UPDATE). Les entités nominales, sans attribut schema, restent résolues dans public via le search_path par défaut
SéquenceRequête native qualifiée : nextval('corpo.corpo_request_ref_seq') — indépendante du search_path
ddl-auto: updateCompatible : Hibernate crée/valide la table dans le bon schéma grâce à l'attribut. Mais le schéma lui-même doit exister avant le démarrage — d'où le CREATE SCHEMA du script ci-dessous, à exécuter en premier
DroitsRien à faire : l'app se connecte en postgres (superuser, propriétaire du schéma créé). Les GRANT commentés en fin de script ne serviront que si un jour un utilisateur applicatif dédié est mis en place
pgAdminLa table apparaîtra sous qlick_flow → Schemas → corpo → Tables (et non sous public)
Flyway (futur, annexe v2)Le jour venu : spring.flyway.schemas: public, corpo et ce script devient un fichier V-numéroté — rien d'autre

Exécution : pgAdmin → base qlick_flow → clic droit → Query Tool → coller → Exécuter (F5). À faire avant le premier démarrage post-implémentation : le schéma doit exister, la séquence est indispensable (ddl-auto n'en crée jamais), et le script garantit défauts et contraintes CHECK que ddl-auto ne poserait pas.

-- ============================================================================
-- Jira AER_LM-257 — Modele de donnees "Demandes Corpo"
-- Architecture a DEUX SCHEMAS dans la meme base qlick_flow :
--   - schema "public" : tables du nominal (cash_request, currency, ...)
--   - schema "corpo"  : tables du perimetre Corporate (isole)
-- A executer dans pgAdmin (Query Tool sur qlick_flow) AVANT le premier
-- demarrage : la SEQUENCE est indispensable (ddl-auto n'en cree jamais)
-- et le SCHEMA doit exister pour qu'Hibernate puisse y creer/valider la table.
-- ============================================================================

-- 1. Le schema dedie
CREATE SCHEMA IF NOT EXISTS corpo;

-- 2. La table des demandes Corporate (strictement les champs de la Jira)
CREATE TABLE corpo.corpo_cash_request (
    id                 uuid PRIMARY KEY,
    creation_date      timestamp,                        -- "Date demande"
    modification_date  timestamp,                        -- "Date mise a jour demande"
    unique_request_ref varchar(30) UNIQUE,               -- "Ref demande" (CO-AAAA-NNNNNN)
    request_type       varchar(20) NOT NULL,             -- ALIMENTATION / RAMASSAGE
    client_ref         varchar(50),                      -- "Ref du client" (zone libre)
    city               varchar(100),                     -- "Ville" (zone libre)
    address_complement varchar(255),                     -- "Complement adresse" (zone libre)
    account_number     varchar(34),                      -- "Num compte" (zone libre)

    -- "Nombre de billets / pieces / centimes ... Valeur par defaut 0"
    nb_banknote_200    integer NOT NULL DEFAULT 0,
    nb_banknote_100    integer NOT NULL DEFAULT 0,
    nb_banknote_50     integer NOT NULL DEFAULT 0,
    nb_banknote_20     integer NOT NULL DEFAULT 0,
    nb_coin_10_dh      integer NOT NULL DEFAULT 0,
    nb_coin_5_dh       integer NOT NULL DEFAULT 0,
    nb_coin_1_dh       integer NOT NULL DEFAULT 0,
    nb_cent_50         integer NOT NULL DEFAULT 0,
    nb_cent_20         integer NOT NULL DEFAULT 0,
    nb_cent_10         integer NOT NULL DEFAULT 0,
    nb_cent_5          integer NOT NULL DEFAULT 0,

    -- "Montant de la demande. Valeur par defaut 0"
    amount             numeric(15,2) NOT NULL DEFAULT 0,

    -- "Flag urgent (oui/non). Valeur par defaut non"
    is_urgent          varchar(3) NOT NULL DEFAULT 'NO',

    CONSTRAINT chk_corpo_request_type CHECK (request_type IN ('ALIMENTATION', 'RAMASSAGE')),
    CONSTRAINT chk_corpo_is_urgent   CHECK (is_urgent IN ('YES', 'NO'))
);

-- 3. Sequence de la reference lisible (CO-2026-000001), dans le schema corpo
CREATE SEQUENCE corpo.corpo_request_ref_seq START WITH 1 INCREMENT BY 1;

package com.bmci.q360.domain.corpo;

import com.bmci.q360.domain.corpo.enums.CorpoRequestType;
import com.bmci.q360.domain.corpo.valueobjects.CorpoCashRequestId;
import com.bmci.q360.domain.entity.common.AggregateRoot;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

import java.math.BigDecimal;
import java.time.LocalDateTime;

/**
 * Agregat "demande Corporate" — Jira AER_LM-257, champs strictement calques
 * sur la Jira : reference, dates, type, identite client (zones libres),
 * 11 compteurs de coupures MAD a defaut 0, montant a defaut 0, flag urgent
 * a defaut NO. Les valeurs faciales sont intrinseques au modele (les 11
 * coupures sont figees par la Jira), aucune resolution de referentiel.
 *
 * Montant : "Valeur par defaut 0 - en cas de ramassage, il sera alimente en
 * direct" (Jira). Interpretation retenue : pour un RAMASSAGE le montant est
 * pris tel que fourni (defaut 0) ; pour une ALIMENTATION il est deduit de la
 * ventilation (somme valeur faciale x quantite). A confirmer avec le metier.
 */
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class CorpoCashRequest extends AggregateRoot<CorpoCashRequestId> {

    private String uniqueRequestRef;
    private CorpoRequestType requestType;
    private String clientRef;
    private String city;
    private String addressComplement;
    private String accountNumber;

    private Integer nbBanknote200 = 0;
    private Integer nbBanknote100 = 0;
    private Integer nbBanknote50 = 0;
    private Integer nbBanknote20 = 0;
    private Integer nbCoin10Dh = 0;
    private Integer nbCoin5Dh = 0;
    private Integer nbCoin1Dh = 0;
    private Integer nbCent50 = 0;
    private Integer nbCent20 = 0;
    private Integer nbCent10 = 0;
    private Integer nbCent5 = 0;

    private BigDecimal amount = BigDecimal.ZERO;
    private String isUrgent = "NO";
    private LocalDateTime createdAt;
    private LocalDateTime updatedAt;

    private static final BigDecimal V_200 = new BigDecimal("200.00");
    private static final BigDecimal V_100 = new BigDecimal("100.00");
    private static final BigDecimal V_50 = new BigDecimal("50.00");
    private static final BigDecimal V_20 = new BigDecimal("20.00");
    private static final BigDecimal V_10 = new BigDecimal("10.00");
    private static final BigDecimal V_5 = new BigDecimal("5.00");
    private static final BigDecimal V_1 = new BigDecimal("1.00");
    private static final BigDecimal V_0_50 = new BigDecimal("0.50");
    private static final BigDecimal V_0_20 = new BigDecimal("0.20");
    private static final BigDecimal V_0_10 = new BigDecimal("0.10");
    private static final BigDecimal V_0_05 = new BigDecimal("0.05");

    /**
     * Applique les valeurs par defaut de la Jira (compteurs 0, montant 0,
     * urgent NO) et alimente le montant d'une ALIMENTATION depuis la
     * ventilation. Le type est le seul champ indispensable (colonne NOT NULL
     * + pivot du calcul).
     */
    public void normalizeAndComputeAmount() {
        if (requestType == null) {
            throw new IllegalStateException(
                    "Le type de demande (ALIMENTATION/RAMASSAGE) est obligatoire");
        }
        normalizeCounters();

        if (requestType == CorpoRequestType.ALIMENTATION) {
            this.amount = computeAmountFromCounters();
        } else if (this.amount == null) {
            this.amount = BigDecimal.ZERO; // defaut Jira, "alimente en direct" sinon
        }

        if (isUrgent == null) {
            isUrgent = "NO"; // defaut Jira
        }
    }

    /** "Valeur par defaut 0" pour chaque compteur (Jira). */
    public void normalizeCounters() {
        nbBanknote200 = zeroIfNull(nbBanknote200);
        nbBanknote100 = zeroIfNull(nbBanknote100);
        nbBanknote50 = zeroIfNull(nbBanknote50);
        nbBanknote20 = zeroIfNull(nbBanknote20);
        nbCoin10Dh = zeroIfNull(nbCoin10Dh);
        nbCoin5Dh = zeroIfNull(nbCoin5Dh);
        nbCoin1Dh = zeroIfNull(nbCoin1Dh);
        nbCent50 = zeroIfNull(nbCent50);
        nbCent20 = zeroIfNull(nbCent20);
        nbCent10 = zeroIfNull(nbCent10);
        nbCent5 = zeroIfNull(nbCent5);
    }

    public BigDecimal computeAmountFromCounters() {
        normalizeCounters();
        return V_200.multiply(bd(nbBanknote200))
                .add(V_100.multiply(bd(nbBanknote100)))
                .add(V_50.multiply(bd(nbBanknote50)))
                .add(V_20.multiply(bd(nbBanknote20)))
                .add(V_10.multiply(bd(nbCoin10Dh)))
                .add(V_5.multiply(bd(nbCoin5Dh)))
                .add(V_1.multiply(bd(nbCoin1Dh)))
                .add(V_0_50.multiply(bd(nbCent50)))
                .add(V_0_20.multiply(bd(nbCent20)))
                .add(V_0_10.multiply(bd(nbCent10)))
                .add(V_0_05.multiply(bd(nbCent5)));
    }

    private Integer zeroIfNull(Integer value) {
        return value == null ? 0 : value;
    }

    private BigDecimal bd(Integer value) {
        return BigDecimal.valueOf(value);
    }
}

package com.bmci.q360.domain.corpo;

import com.bmci.q360.domain.corpo.enums.CorpoRequestType;
import org.junit.jupiter.api.Assertions;
import org.junit.jupiter.api.Test;

import java.math.BigDecimal;

/**
 * Tests du modele corpo (Jira AER_LM-257) — module fund-management-domain-core.
 */
public class CorpoCashRequestTest {

    @Test
    void shouldComputeAmountFromCounters() {
        CorpoCashRequest request = new CorpoCashRequest();
        request.setNbBanknote200(2);   // 400.00
        request.setNbBanknote100(3);   // 300.00
        request.setNbCent50(4);        //   2.00

        BigDecimal amount = request.computeAmountFromCounters();

        Assertions.assertEquals(0, new BigDecimal("702.00").compareTo(amount));
    }

    @Test
    void alimentationShouldTakeAmountFromCounters() {
        CorpoCashRequest request = new CorpoCashRequest();
        request.setRequestType(CorpoRequestType.ALIMENTATION);
        request.setNbBanknote200(10); // 2000.00
        request.setAmount(new BigDecimal("999.99")); // remplace par le calcul

        request.normalizeAndComputeAmount();

        Assertions.assertEquals(0, new BigDecimal("2000.00").compareTo(request.getAmount()));
    }

    @Test
    void ramassageShouldKeepProvidedAmountAndDefaultToZero() {
        CorpoCashRequest provided = new CorpoCashRequest();
        provided.setRequestType(CorpoRequestType.RAMASSAGE);
        provided.setAmount(new BigDecimal("15000.00")); // "alimente en direct"
        provided.normalizeAndComputeAmount();
        Assertions.assertEquals(0, new BigDecimal("15000.00").compareTo(provided.getAmount()));

        CorpoCashRequest empty = new CorpoCashRequest();
        empty.setRequestType(CorpoRequestType.RAMASSAGE);
        empty.setAmount(null);
        empty.normalizeAndComputeAmount();
        Assertions.assertEquals(0, BigDecimal.ZERO.compareTo(empty.getAmount())); // defaut Jira
    }

    @Test
    void missingTypeShouldBeRejected() {
        CorpoCashRequest request = new CorpoCashRequest();

        IllegalStateException exception = Assertions.assertThrows(
                IllegalStateException.class, request::normalizeAndComputeAmount);

        Assertions.assertTrue(exception.getMessage().contains("type de demande"));
    }

    @Test
    void jiraDefaultsShouldBeApplied() {
        CorpoCashRequest request = new CorpoCashRequest();
        request.setRequestType(CorpoRequestType.RAMASSAGE);
        request.setNbBanknote200(null); // compteur non fourni -> 0
        request.setIsUrgent(null);      // urgent non fourni -> NO

        request.normalizeAndComputeAmount();

        Assertions.assertEquals(0, request.getNbBanknote200());
        Assertions.assertEquals("NO", request.getIsUrgent());
    }
}

/**
 * Adaptateur de persistance des demandes Corporate — Jira AER_LM-257.
 * "Ref demande" generee a la creation via la sequence corpo_request_ref_seq
 * (CO-AAAA-NNNNNN). Erreurs "not found" en RuntimeException, comme le nominal.
 */
@Component
@Slf4j
@RequiredArgsConstructor
public class CorpoDemandRepositoryImpl implements CorpoDemandRepository {

    private final CorpoCashRequestJpaRepository corpoJpaRepository;
    private final CorpoCashRequestPersistenceMapper mapper;

    @Override
    @Transactional
    public CorpoCashRequest save(CorpoCashRequest request) {
        if (request.getUniqueRequestRef() == null) {
            Long next = corpoJpaRepository.nextRequestRef();
            request.setUniqueRequestRef(
                    "CO-" + Year.now().getValue() + "-" + String.format("%06d", next));
        }
        CorpoCashRequestEntity entity = mapper.toPersistence(request);
        return mapper.toDomain(corpoJpaRepository.save(entity));
    }

    @Override
    public CorpoCashRequest getById(UUID id) {
        CorpoCashRequestEntity entity = corpoJpaRepository.findById(id)
                .orElseThrow(() -> new RuntimeException("Corpo cash demand not found"));
        return mapper.toDomain(entity);
    }

    @Override
    public List<CorpoCashRequest> findAll() {
        return corpoJpaRepository.findAll().stream().map(mapper::toDomain).toList();
    }

    @Override
    @Transactional
    public CorpoCashRequest update(UUID id, CorpoCashRequest request) {
        CorpoCashRequestEntity existing = corpoJpaRepository.findById(id)
                .orElseThrow(() -> new RuntimeException("Corpo cash demand not found"));
        mapper.applyFields(existing, request);
        return mapper.toDomain(corpoJpaRepository.save(existing));

/**

@ExtendWith(MockitoExtension.class)
public class CorpoDemandRepositoryTest {

    @Mock
    private CorpoCashRequestJpaRepository corpoJpaRepository;

    @Mock
    private CorpoCashRequestPersistenceMapper mapper;

    @InjectMocks
    private CorpoDemandRepositoryImpl corpoDemandRepository;

    private UUID requestId;

    @BeforeEach
    void setUp() {
        requestId = UUID.randomUUID();
    }

    @Test
    void saveShouldGenerateReadableReferenceWhenAbsent() {
        CorpoCashRequest domain = new CorpoCashRequest(); // pas de reference
        CorpoCashRequestEntity entity = new CorpoCashRequestEntity();

        Mockito.when(corpoJpaRepository.nextRequestRef()).thenReturn(42L);
        Mockito.when(mapper.toPersistence(domain)).thenReturn(entity);
        Mockito.when(corpoJpaRepository.save(entity)).thenReturn(entity);
        Mockito.when(mapper.toDomain(entity)).thenReturn(domain);

        corpoDemandRepository.save(domain);

        Assertions.assertEquals(
                "CO-" + Year.now().getValue() + "-000042", domain.getUniqueRequestRef());
    }

    @Test
    void saveShouldKeepExistingReference() {
        CorpoCashRequest domain = new CorpoCashRequest();
        domain.setUniqueRequestRef("CO-2026-000007");
        CorpoCashRequestEntity entity = new CorpoCashRequestEntity();

        Mockito.when(mapper.toPersistence(domain)).thenReturn(entity);
        Mockito.when(corpoJpaRepository.save(entity)).thenReturn(entity);
        Mockito.when(mapper.toDomain(entity)).thenReturn(domain);

        corpoDemandRepository.save(domain);

        Assertions.assertEquals("CO-2026-000007", domain.getUniqueRequestRef());
        Mockito.verify(corpoJpaRepository, Mockito.never()).nextRequestRef();
    }

    @Test
    void getByIdShouldThrowWhenAbsent() {
        Mockito.when(corpoJpaRepository.findById(requestId)).thenReturn(Optional.empty());

        RuntimeException exception = Assertions.assertThrows(
                RuntimeException.class,
                () -> corpoDemandRepository.getById(requestId));

        Assertions.assertEquals("Corpo cash demand not found", exception.getMessage());
    }

    @Test
    void updateShouldApplyFieldsOnExistingEntity() {
        CorpoCashRequestEntity existing = new CorpoCashRequestEntity();
        existing.setId(requestId);
        CorpoCashRequest incoming = new CorpoCashRequest();
        CorpoCashRequest expected = new CorpoCashRequest();

        Mockito.when(corpoJpaRepository.findById(requestId)).thenReturn(Optional.of(existing));
        Mockito.when(corpoJpaRepository.save(existing)).thenReturn(existing);
        Mockito.when(mapper.toDomain(existing)).thenReturn(expected);

        CorpoCashRequest result = corpoDemandRepository.update(requestId, incoming);

        Mockito.verify(mapper).applyFields(existing, incoming);
        Assertions.assertSame(expected, result);
    }
